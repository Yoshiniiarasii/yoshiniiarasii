        <script>
        var highlightLayer;
        function highlightFeature(e) {
            highlightLayer = e.target;
            highlightLayer.openPopup();
        }

        // 1. INITIALIZE MAP & GOOGLE SATELLITE
        var map = L.map('map', {
            zoomControl: false, // Custom position set below
            maxZoom: 28, 
            minZoom: 1
        }).fitBounds([[2.102458566417635,102.52039111454198],[2.104151315938527,102.52385671214154]]);

        var googleSat = L.tileLayer('https://mt1.google.com/vt/lyrs=s&x={x}&y={y}&z={z}', {
            attribution: '&copy; Google Maps'
        }).addTo(map);

        var hash = new L.Hash(map);
        var autolinker = new Autolinker({truncate: {length: 30, location: 'smart'}});

        function removeEmptyRowsFromPopupContent(content, feature) {
            var tempDiv = document.createElement('div');
            tempDiv.innerHTML = content;
            var rows = tempDiv.querySelectorAll('tr');
            for (var i = 0; i < rows.length; i++) {
                var td = rows[i].querySelector('td.visible-with-data');
                var key = td ? td.id : '';
                if (td && td.classList.contains('visible-with-data') && feature.properties[key] == null) {
                    rows[i].parentNode.removeChild(rows[i]);
                }
            }
            return tempDiv.innerHTML;
        }

        function addClassToPopupIfMedia(content, popup) {
            var tempDiv = document.createElement('div');
            tempDiv.innerHTML = content;
            var imgTd = tempDiv.querySelector('td img');
            if (imgTd) {
                var src = imgTd.getAttribute('src');
                if (/\.(jpg|jpeg|png|gif|bmp|webp|avif)$/i.test(src)) {
                    popup._contentNode.classList.add('media');
                    setTimeout(function() { popup.update(); }, 10);
                }
            }
        }

        // 2. ZOOM CONTROL
        L.control.zoom({ position: 'topleft' }).addTo(map);

        var bounds_group = new L.featureGroup([]);

        // 3. TRANSPARENT LOT LAYER STYLE
        function pop_LOTFINAL_0(feature, layer) {
            layer.on({
                mouseout: function(e) { if (typeof layer.closePopup == 'function') { layer.closePopup(); } },
                mouseover: highlightFeature,
            });
            var popupContent = '<table>\
                    <tr><td class="visible-with-data" id="LOT" colspan="2"><strong>LOT: </strong>' + (feature.properties['LOT'] !== null ? autolinker.link(String(feature.properties['LOT'])) : '') + '</td></tr>\
                    <tr><td colspan="2"><strong>Area: </strong>' + (feature.properties['AREA'] !== null ? autolinker.link(String(feature.properties['AREA'])) : '') + '</td></tr>\
                </table>';
            var content = removeEmptyRowsFromPopupContent(popupContent, feature);
            layer.on('popupopen', function(e) { addClassToPopupIfMedia(content, e.popup); });
            layer.bindPopup(content, { maxHeight: 400 });
        }

        function style_LOTFINAL_0_0() {
            return {
                pane: 'pane_LOTFINAL_0',
                opacity: 1,
                color: 'rgba(255,255,0,1.0)', // Yellow borders for visibility
                dashArray: '',
                lineCap: 'butt',
                lineJoin: 'miter',
                weight: 2.0, 
                fill: true,
                fillOpacity: 0.0, // FULLY TRANSPARENT
                interactive: true,
            }
        }

        map.createPane('pane_LOTFINAL_0');
        map.getPane('pane_LOTFINAL_0').style.zIndex = 400;

        var layer_LOTFINAL_0 = new L.geoJson(json_LOTFINAL_0, {
            interactive: true,
            dataVar: 'json_LOTFINAL_0',
            layerName: 'layer_LOTFINAL_0',
            pane: 'pane_LOTFINAL_0',
            onEachFeature: pop_LOTFINAL_0,
            style: style_LOTFINAL_0_0,
        });
        bounds_group.addLayer(layer_LOTFINAL_0);
        map.addLayer(layer_LOTFINAL_0);

        // 4. POINT LAYER STYLE (TANDASEMPADANGABUNG)
        function style_TANDASEMPADANGABUNG_1_0() {
            return {
                pane: 'pane_TANDASEMPADANGABUNG_1',
                radius: 4.0,
                opacity: 1,
                color: 'rgba(255,255,255,1.0)',
                weight: 1,
                fill: true,
                fillOpacity: 1,
                fillColor: 'rgba(190,207,80,1.0)',
                interactive: true,
            }
        }
        map.createPane('pane_TANDASEMPADANGABUNG_1');
        map.getPane('pane_TANDASEMPADANGABUNG_1').style.zIndex = 401;

        var layer_TANDASEMPADANGABUNG_1 = new L.geoJson(json_TANDASEMPADANGABUNG_1, {
            interactive: true,
            dataVar: 'json_TANDASEMPADANGABUNG_1',
            layerName: 'layer_TANDASEMPADANGABUNG_1',
            pane: 'pane_TANDASEMPADANGABUNG_1',
            pointToLayer: function (feature, latlng) {
                return L.circleMarker(latlng, style_TANDASEMPADANGABUNG_1_0());
            },
        });
        bounds_group.addLayer(layer_TANDASEMPADANGABUNG_1);
        map.addLayer(layer_TANDASEMPADANGABUNG_1);

        // 5. SEARCH WITH ZOOM-IN FUNCTIONALITY
        var searchControl = new L.Control.Search({
            layer: layer_LOTFINAL_0,
            initial: false,
            hideMarkerOnCollapse: true,
            propertyName: 'LOT',
            moveToLocation: function(latlng, title, map) {
                map.setView(latlng, 19); // Zooms in close to the searched lot
            }
        });
        map.addControl(searchControl);

        // Set binocular icon for search
        document.getElementsByClassName('search-button')[0].className += ' fa fa-binoculars';

        </script>
